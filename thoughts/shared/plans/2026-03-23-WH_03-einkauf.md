```sql
-- 1. Create enums
CREATE TYPE wh_purchase_order_status AS ENUM (
  'DRAFT',
  'ORDERED',
  'PARTIALLY_RECEIVED',

  'CANCELLED'

  'PHONE',
  'EMAIL',
  'FAX',
  'PRINT'
);

CREATE TABLE wh_purchase_orders (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
  number VARCHAR(50) NOT NULL,
  supplier_id UUID NOT NULL REFERENCES crm_addresses(id),
  contact_id UUID REFERENCES crm_contacts(id) ON DELETE SET NULL,
  inquiry_id UUID REFERENCES crm_inquiries(id) ON DELETE SET NULL,
  status wh_purchase_order_status NOT NULL DEFAULT 'DRAFT',
  order_date TIMESTAMPTZ(6),
  requested_delivery TIMESTAMPTZ(6),
  confirmed_delivery TIMESTAMPTZ(6),
  order_method wh_purchase_order_method,
  order_method_note TEXT,
  notes TEXT,
  subtotal_net DOUBLE PRECISION NOT NULL DEFAULT 0,
  total_gross DOUBLE PRECISION NOT NULL DEFAULT 0,
  printed_at TIMESTAMPTZ(6),
  created_at TIMESTAMPTZ(6) NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ(6) NOT NULL DEFAULT now(),
  created_by_id UUID,
  CONSTRAINT uq_wh_purchase_orders_tenant_number UNIQUE(tenant_id, number)
);

-- 3. Create purchase order positions table
CREATE TABLE wh_purchase_order_positions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  purchase_order_id UUID NOT NULL REFERENCES wh_purchase_orders(id) ON DELETE CASCADE,
  sort_order INT NOT NULL,
  article_id UUID NOT NULL REFERENCES wh_articles(id),
  supplier_article_number VARCHAR(100),
  description TEXT,
  quantity DOUBLE PRECISION NOT NULL,
  received_quantity DOUBLE PRECISION NOT NULL DEFAULT 0,
  unit VARCHAR(20),
  unit_price DOUBLE PRECISION,
  flat_costs DOUBLE PRECISION,
  total_price DOUBLE PRECISION,
  requested_delivery TIMESTAMPTZ(6),
  confirmed_delivery TIMESTAMPTZ(6),
  created_at TIMESTAMPTZ(6) NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ(6) NOT NULL DEFAULT now()
);

-- 4. Create indexes
CREATE INDEX idx_wh_purchase_orders_tenant_status ON wh_purchase_orders(tenant_id, status);
CREATE INDEX idx_wh_purchase_orders_tenant_supplier ON wh_purchase_orders(tenant_id, supplier_id);
CREATE INDEX idx_wh_purchase_orders_tenant_delivery ON wh_purchase_orders(tenant_id, requested_delivery);
CREATE INDEX idx_wh_purchase_order_positions_order_sort ON wh_purchase_order_positions(purchase_order_id, sort_order);
```

| Function | Signature | Logic |
|----------|-----------|-------|
| `list` | `(prisma, tenantId, params)` | Delegates to `repo.findMany`. Returns `{ items, total }`. |
| `getById` | `(prisma, tenantId, id)` | Calls `repo.findById`. Throws `WhPurchaseOrderNotFoundError` if null. |

| `update` | `(prisma, tenantId, input: { id, supplierId?, contactId?, requestedDelivery?, confirmedDelivery?, notes?, inquiryId? }, audit?)` | 1. `getById` to fetch existing. 2. Reject if `status !== "DRAFT"` with `WhPurchaseOrderValidationError("Can only edit draft purchase orders")`. 3. If supplierId changed, validate new supplier. 4. `repo.update(prisma, tenantId, id, data)`. 5. Audit log with diff. |
| `delete` | `(prisma, tenantId, id, audit?)` | 1. `getById`. 2. Reject if `status !== "DRAFT"`. 3. `repo.softDeleteById(prisma, tenantId, id)`. Throws `WhPurchaseOrderNotFoundError` if count=0. 4. Audit log. |
| `sendOrder` | `(prisma, tenantId, id, input: { method, methodNote? }, audit?)` | 1. `getById`. 2. Reject if `status !== "DRAFT"`. 3. Reject if no positions (`positions.length === 0`). 4. `repo.update(prisma, tenantId, id, { status: "ORDERED", orderDate: new Date(), orderMethod: input.method, orderMethodNote: input.methodNote })`. 5. Audit log. |
| `cancel` | `(prisma, tenantId, id, audit?)` | 1. `getById`. 2. Reject if status is `RECEIVED` or `CANCELLED`. 3. `repo.update(prisma, tenantId, id, { status: "CANCELLED" })`. 4. Audit log. |
| `addPosition` | `(prisma, tenantId, input: { purchaseOrderId, articleId, quantity, unitPrice?, unit?, description?, flatCosts?, requestedDelivery?, confirmedDelivery? }, audit?)` | 1. `getById(prisma, tenantId, input.purchaseOrderId)` — verifies tenant + gets order. 2. Reject if `status !== "DRAFT"`. 3. Validate article exists (`prisma.whArticle.findFirst({ where: { id: input.articleId, tenantId } })`). 4. Auto-fill from `WhArticleSupplier`: look up `prisma.whArticleSupplier.findFirst({ where: { articleId: input.articleId, supplierId: order.supplierId } })`. If found, default `supplierArticleNumber`, `unitPrice` (from supplier.buyPrice), `unit` (from supplier.orderUnit). If not found, fall back to article's `buyPrice` and `unit`. 5. Calculate `sortOrder = await repo.countPositions(prisma, input.purchaseOrderId)`. 6. Calculate `totalPrice = (quantity * unitPrice) + (flatCosts ?? 0)`. 7. `repo.createPosition(prisma, input.purchaseOrderId, { ...data })`. 8. Recalculate totals: `recalculateTotals(prisma, tenantId, input.purchaseOrderId)`. 9. Audit log. |
| `updatePosition` | `(prisma, tenantId, input: { id, quantity?, unitPrice?, unit?, description?, flatCosts?, requestedDelivery?, confirmedDelivery? }, audit?)` | 1. Verify position exists and order is tenant-scoped and DRAFT. 2. Recalculate `totalPrice`. 3. `repo.updatePosition(prisma, tenantId, input.id, data)`. 4. `recalculateTotals`. 5. Audit log. |
| `deletePosition` | `(prisma, tenantId, positionId, audit?)` | 1. Verify position + order tenant + DRAFT status. 2. `repo.deletePosition(prisma, tenantId, positionId)`. 3. `recalculateTotals`. 4. Audit log. |
| `listPositions` | `(prisma, tenantId, purchaseOrderId)` | `repo.findPositionsByOrder(prisma, tenantId, purchaseOrderId)` |
| `getReorderSuggestions` | `(prisma, tenantId, supplierId?)` | `repo.findArticlesBelowMinStock(prisma, tenantId, supplierId)`. Post-process each article: calculate `suggestedQty = Math.max((minStock - currentStock), defaultOrderQty ?? 0)`. Sort by urgency: `(currentStock / minStock)` ascending. Return `{ articleId, articleNumber, articleName, currentStock, minStock, deficit, supplierId, supplierName, supplierArticleNumber, suggestedQty, unitPrice }[]`. |
| `createFromSuggestions` | `(prisma, tenantId, input: { supplierId, articleIds: string[] }, createdById, audit?)` | 1. Validate supplier. 2. Create DRAFT PO via `create()`. 3. For each articleId: call `addPosition()` with calculated suggestedQty. 4. Return the PO with positions. |

**Helper:**

```ts
async function recalculateTotals(prisma: PrismaClient, tenantId: string, purchaseOrderId: string) {
  const positions = await prisma.whPurchaseOrderPosition.findMany({
    where: { purchaseOrderId },
    select: { totalPrice: true },
  })
  const subtotalNet = positions.reduce((sum, p) => sum + (p.totalPrice ?? 0), 0)
  await prisma.whPurchaseOrder.updateMany({
    where: { id: purchaseOrderId, tenantId },
    data: { subtotalNet, totalGross: subtotalNet },
  })
}
```
